#calculator with 10 functions:

add <- function(num1, num2) {
    return(num1 + num2)
}

subtract <- function(num1, num2) {
    return(num1 - num2)
}

multiply <- function(num1, num2) {
    return(num1 * num2)
}

divide <- function(num1, num2) {
  if (num2 == 0) {
    print ("You shall not divide by 0")
  } else {
    return(num1 / num2)
  }
}

exponent <- function(num1, num2) {
  return (num1 ^ num2)
}

factorial <- function(x){
y <- 1
for(i in 1:x){
y <-y*((1:x)[i])
}
print(y)
}

squareroot <- function(num1) {
  if (num1 >= 0) {
       sqrt(num1)
    } else {
       print (paste("The square root of a negative is undefined over the Reals")) 
    }
}

cosine <- function (num1) {
  cos(num1)
}

sine <- function (num1) {
  sin(num1)
}

tangent <- function (num1) {
  tan(num1)
}
